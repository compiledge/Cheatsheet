\begin{ppbox}{\faTerminal}{Start program}
	\command{> gdb $[bin]$}{start GDB}
	\flag{-tui}{show code}
	\flag{-command}{read init file}
\end{ppbox}

\begin{obox}{\faStopCircleO}{Break points}
	\command{> break $i$}{insert bk point at $i$}
	\command{> condition $i$ $exp$}{insert cond}
	\command{> clear $i$}{remove bk}
\end{obox}

\begin{gbox}{\faPlayCircle}{Flux Control}
	\command{> run $[args]$}{start program}
	\command{> continue}{resume execution}
	\command{> next}{step over}
	\command{> step}{step info}
\end{gbox}

\begin{bbox}{\faSearch}{Inspecting Variables}
	\command{> print $i$}{print var $i$}
\end{bbox}

\begin{dbox}{\faSortNumericDesc}{Stacks}
	\command{> frame $i$}{examine stack $i$}
	\command{> backtrace}{show entire stack}
	\command{> up}{show parent stack}
	\command{> down}{show child stack}
\end{dbox}
